cd /nfs/users/nfs_f/fc4/lustre_scratch118/67.mtb_within_host/tb-profiler

paths_file="/nfs/users/nfs_f/fc4/lustre_scratch118/67.mtb_within_host/datasets/mtb_wh_extended.v1.run_accessions.lane_ids.fastq_paths.on_farm.1.txt";

laSamples=`awk -F'\t' '{ print $1}' $paths_file`;

cd $results_dir

for sample in $laSamples
do

echo $sample

out_report2=$results_dir$sample".results.json";

if [ -f $out_report2  ]
then

	bsub -q normal -G team81 -J $sample"_tb-profiler_rf" -o $sample"_tb-profiler_rf.out" -R "select[mem > 2000] rusage[mem=2000]" -M 2000 "tb-profiler reformat $out_report2 --txt --csv --dir $results_dir"

else
	echo $out_report2" not found!"
fi
done
